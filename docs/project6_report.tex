{\color{GREEN}T}{his} project studies the response of a third-order Butterworth lowpass filter to a periodic rectangular pulse train. The work has two connected parts. First, the filter is analyzed using its frequency response and Fourier series. Second, the same physical circuit is simulated in \simulink{} using Simscape Electrical. The final comparison checks whether the frequency-domain reconstruction and circuit simulation produce the same output waveform.

\sectionheader{Setup: Filter Circuit and Transfer Function}

\begin{noteBox}
The filter is a passive third-order lowpass RLC network. The input voltage is \(x(t)\), and the output voltage \(y(t)\) is measured across the \(100\ \Omega\) resistor.
\end{noteBox}

\end{circuitikz}

The transfer function is given by
\[
H(s)=\frac{Y(s)}{X(s)}
=
\frac{1}{(s/\beta)^3+2(s/\beta)^2+2(s/\beta)+1},
\]
where
\[
\beta=\frac{50}{3}\ \text{rad/s}.
\]

It is often more convenient for \matlab{} to rewrite this transfer function as an ordinary polynomial in \(s\). Multiply the denominator by \(\beta^3\):
\[
H(s)
=
\frac{\beta^3}
{s^3+2\beta s^2+2\beta^2s+\beta^3}.
\]
Therefore the numerator and denominator coefficient arrays are
\[
\texttt{num}=\beta^3,
\]
and
\[
\texttt{den}
=
\begin{bmatrix}
1 & 2\beta & 2\beta^2 & \beta^3
\end{bmatrix}.
\]

The input pulse train has period
\[
T_0=1.0\ \text{s},
\]
duty cycle
\[
\alpha=0.40,
\]
and fundamental angular frequency
\[
\Omega_0=\frac{2\pi}{T_0}=2\pi\ \text{rad/s}.
\]

\sectionheader{Problem (i): Frequency Response}

\begin{problem}{i}
Using the \texttt{polyval} function, evaluate the complex frequency response \(H(\jj\Omega)\) for
\[
\Omega = 0:(0.1)\Omega_0:(100)\Omega_0,
\]
where
\[
\Omega_0=2\pi\ \text{rad/s}.
\]
Plot the magnitude response \(|H(\jj\Omega)|\) in dB and the phase response \(\angle H(\jj\Omega)\) as subplots on the same figure. Confirm that
\[
\Omega=\beta=\frac{50}{3}\ \text{rad/s}
\]
is the \(-3\text{ dB}\) point.
\end{problem}

\begin{solution}{i}
The frequency response is found by substituting
\[
s=\jj\Omega
\]
into \(H(s)\). Thus
\[
H(\jj\Omega)
=
\frac{1}{(\jj\Omega/\beta)^3+2(\jj\Omega/\beta)^2+2(\jj\Omega/\beta)+1}.
\]

Equivalently,
\[
H(\jj\Omega)
=
\frac{\beta^3}{(\jj\Omega)^3+2\beta(\jj\Omega)^2+2\beta^2(\jj\Omega)+\beta^3}.
\]

The magnitude in decibels is
\[
|H(\jj\Omega)|_{\text{dB}}
=
20\log_{10}|H(\jj\Omega)|.
\]
The phase is
\[
\angle H(\jj\Omega)=\arg(H(\jj\Omega)).
\]

At the cutoff frequency,
\[
\Omega=\beta,
\]
we compute
\[
H(\jj\beta)
=
\frac{1}{(\jj)^3+2(\jj)^2+2(\jj)+1}.
\]
Since
\[
\jj^2=-1,
\qquad
\jj^3=-\jj,
\]
the denominator becomes
\[
-\jj+2(-1)+2\jj+1
=
-2+1+\jj
=
-1+\jj.
\]
Therefore
\[
H(\jj\beta)
=
\frac{1}{-1+\jj}.
\]
Its magnitude is
\[
|H(\jj\beta)|
=
\frac{1}{|-1+\jj|}
=
\frac{1}{\sqrt{(-1)^2+1^2}}
=
\frac{1}{\sqrt{2}}.
\]
In dB,
\[
20\log_{10}\left(\frac{1}{\sqrt{2}}\right)
=
-3.0103\ \text{dB}.
\]
Thus
\[
\boxed{
\Omega=\beta=\frac{50}{3}\ \text{rad/s}
}
\]
is the \(-3\text{ dB}\) point.

\subsection*{MATLAB Code for Part (i)}

\begin{codeBox}
\begin{verbatim}
clear; clc; close all;

beta = 50/3;
T0 = 1.0;
Omega0 = 2*pi/T0;

Omega = 0:(0.1*Omega0):(100*Omega0);
s = 1j*Omega;

num = beta^3;
den = [1, 2*beta, 2*beta^2, beta^3];

H = polyval(num,s)./polyval(den,s);

mag_dB = 20*log10(abs(H));
phase_deg = unwrap(angle(H))*180/pi;

H_beta = polyval(num,1j*beta)./polyval(den,1j*beta);
mag_beta_dB = 20*log10(abs(H_beta));

fprintf('beta = %.6f rad/s\n', beta);
fprintf('|H(j beta)| = %.8f\n', abs(H_beta));
fprintf('|H(j beta)| dB = %.8f dB\n', mag_beta_dB);

figure('Color','w');
subplot(2,1,1);
plot(Omega,mag_dB,'LineWidth',1.6);
grid on;
xlabel('\Omega rad/s');
ylabel('|H(j\Omega)| dB');
title('Magnitude Response');

subplot(2,1,2);
plot(Omega,phase_deg,'LineWidth',1.6);
grid on;
xlabel('\Omega rad/s');
ylabel('\angle H(j\Omega) degrees');
title('Phase Response');

saveas(gcf,'../figures/part_i_frequency_response.png');
\end{verbatim}
\end{codeBox}

\figmaybe{../figures/part_i_frequency_response.png}{0.92}{Part (i): Magnitude and phase response of the Butterworth lowpass filter.}

\[

}
\qquad
\boxed{

\]
\end{solution}

\sectionheader{Problem (ii): Output Fourier Series}

\begin{problem}{ii}
Assuming unit pulse amplitude and even symmetry about \(t=0\), the periodic input \(x(t)\) is represented by the complex Fourier series
\[
x(t)=\sum_{k=-\infty}^{\infty}X_k\ee^{\jj k\Omega_0t},
\]
where
\[
X_k=\frac{\sin(k\alpha\pi)}{k\pi}.
\]
Write an equation for the complex Fourier series of the output \(y(t)\) in terms of \(X_k\), \(H(\jj\Omega)\), and \(\Omega_0\). Then express \(y(t)\) as a sum of cosines over nonnegative harmonic indices \(k\).
\end{problem}

\begin{solution}{ii}
For an LTI system, each complex exponential input
\[
\ee^{\jj k\Omega_0t}
\]
is an eigenfunction of the system. Therefore, the \(k\)-th harmonic gets multiplied by the frequency response evaluated at that harmonic:
\[
H(\jj k\Omega_0).
\]

The input is
\[
x(t)
=
\sum_{k=-\infty}^{\infty}
X_k\ee^{\jj k\Omega_0t}.
\]
Therefore the output is
\[
\boxed{
y(t)
=
\sum_{k=-\infty}^{\infty}
X_kH(\jj k\Omega_0)
\ee^{\jj k\Omega_0t}.
}
\]

Define
\[
Y_k=X_kH(\jj k\Omega_0).
\]
Then
\[
\boxed{
y(t)=\sum_{k=-\infty}^{\infty}Y_k\ee^{\jj k\Omega_0t}.
}
\]

For \(k=0\), the formula
\[
X_k=\frac{\sin(k\alpha\pi)}{k\pi}
\]
has the removable \(0/0\) form. Taking the limit gives
\[
X_0=\alpha.
\]
Because the DC gain of this filter is
\[
H(0)=1,
\]
the DC output term is
\[
Y_0=X_0H(0)=\alpha.
\]

Now use conjugate symmetry. Since \(x(t)\) is real and even, \(X_k\) is real and
\[
X_{-k}=X_k.
\]
For a real-coefficient LTI system,
\[
H(-\jj k\Omega_0)=H(\jj k\Omega_0)^*.
\]
Thus the negative and positive harmonics pair together into cosines.

For \(k\geq 1\),
\[
X_kH(\jj k\Omega_0)\ee^{\jj k\Omega_0t}
+
X_{-k}H(-\jj k\Omega_0)\ee^{-\jj k\Omega_0t}
\]
becomes
\[
2X_k|H(\jj k\Omega_0)|
\cos\left(k\Omega_0t+\angle H(\jj k\Omega_0)\right).
\]

Therefore,
\[
\boxed{
y(t)
=
\alpha
+
2\sum_{k=1}^{\infty}
X_k
|H(\jj k\Omega_0)|
\cos\left(k\Omega_0t+\angle H(\jj k\Omega_0)\right).
}
\]

Substituting
\[
X_k=\frac{\sin(k\alpha\pi)}{k\pi}
\]
gives
\[
\boxed{
y(t)
=
\alpha
+
2\sum_{k=1}^{\infty}
\frac{\sin(k\alpha\pi)}{k\pi}
|H(\jj k\Omega_0)|
\cos\left(k\Omega_0t+\angle H(\jj k\Omega_0)\right).
}
\]

For this project,
\[
\alpha=0.40,
\qquad
\Omega_0=2\pi.
\]
So
\[
\boxed{
y(t)
=
0.40
+
2\sum_{k=1}^{\infty}
\frac{\sin(0.40k\pi)}{k\pi}
|H(\jj 2\pi k)|
\cos\left(2\pi kt+\angle H(\jj 2\pi k)\right).
}
\]

\begin{noteBox}
It is important not to replace \(X_k\) by \(|X_k|\) unless an additional phase term is included. The coefficient
\[
X_k=\frac{\sin(k\alpha\pi)}{k\pi}
\]
can be positive, zero, or negative. Keeping \(X_k\) with its sign is the cleanest correct expression.
\end{noteBox}
\end{solution}

\sectionheader{Problem (iii): IFFT Reconstruction of \(y(t)\)}

\begin{problem}{iii}
Because of the steep roll-off in \(|H(\jj\Omega)|\), truncate the output Fourier series at
\[
|k|=100.
\]
Use the IFFT function to compute
\[
N=1000
\]
samples of \(y(t)\) over one period \([0,T_0)\).
\end{problem}

\begin{solution}{iii}
The truncated Fourier series is
\[
y(t)
\approx
\sum_{k=-100}^{100}
X_kH(\jj k\Omega_0)\ee^{\jj k\Omega_0t}.
\]

Let
\[
Y_k=X_kH(\jj k\Omega_0).
\]
To use an \(N\)-point IFFT, place the coefficient \(Y_k\) into the FFT bin corresponding to harmonic \(k\). For MATLAB indexing,
\[
\text{index}=\operatorname{mod}(k,N)+1.
\]

MATLAB's IFFT convention is
\[
\texttt{ifft}(Y)[n]
=
\frac{1}{N}
\sum_{m=0}^{N-1}
Y[m]\ee^{\jj 2\pi mn/N}.
\]
However, the Fourier series form is
\[
y[n]
=
\sum_k Y_k\ee^{\jj 2\pi kn/N}.
\]
Therefore, after using \texttt{ifft}, multiply by \(N\):
\[
\boxed{
y[n]=N\cdot\texttt{ifft}(Y).
}
\]

\subsection*{MATLAB Code for Part (iii)}

\begin{codeBox}
\begin{verbatim}
clear; clc; close all;

beta = 50/3;
T0 = 1.0;
alpha = 0.40;
Omega0 = 2*pi/T0;

N = 1000;
Kmax = 100;

Ts = T0/N;
t = (0:N-1)*Ts;

num = beta^3;
den = [1, 2*beta, 2*beta^2, beta^3];

Y = zeros(1,N);

for k = -Kmax:Kmax

    if k == 0
        Xk = alpha;
    else
        Xk = sin(k*alpha*pi)/(k*pi);
    end

    Hk = polyval(num,1j*k*Omega0)./polyval(den,1j*k*Omega0);

    Yk = Xk*Hk;

    idx = mod(k,N) + 1;
    Y(idx) = Yk;
end

y_ifft = real(ifft(Y)*N);

figure('Color','w');
plot(t,y_ifft,'LineWidth',1.6);
grid on;
xlabel('t seconds');
ylabel('y(t)');
title('Output y(t) from Truncated Fourier Series using IFFT');

saveas(gcf,'../figures/part_iii_ifft_response.png');

save('../data/part_iii_output.mat','t','y_ifft','T0','alpha','Omega0','beta');
\end{verbatim}
\end{codeBox}

\figmaybe{../figures/part_iii_ifft_response.png}{0.92}{Part (iii): Output waveform computed using the IFFT of the truncated Fourier coefficients.}

\begin{noteBox}
The output is smoother than the input pulse because the lowpass filter strongly attenuates high harmonics. The sharp edges of the pulse require high-frequency harmonic content, and those harmonics are suppressed by the Butterworth filter.
\end{noteBox}
\end{solution}

\sectionheader{Problem (v): Comparison of Fourier and Simulink Outputs}

\begin{problem}{v}
Plot two periods of \(y(t)\) obtained from Part (iii) as a solid graph. On the same figure, plot two periods of the simulated output obtained from Part (iv) using discrete markers. The two plots should agree.
\end{problem}

\begin{solution}{v}
The Fourier reconstruction from Part (iii) gives one period of the steady-state output. To compare two periods, repeat the one-period vector:
\[
y_{\text{two}}(t)
=
[y(t),y(t)].
\]

The Simscape output contains transients at the beginning, so the last two periods are used. Since the total simulation time is \(10T_0\), take the interval
\[
8T_0\le t\le 10T_0.
\]
Then shift that time interval back to
\[
0\le t\le 2T_0.
\]
