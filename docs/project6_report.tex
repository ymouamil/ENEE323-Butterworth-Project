{\color{GREEN}T}{his} report computes the response of a third-order passive Butterworth lowpass filter to a periodic rectangular pulse train. The solution is split into a frequency-domain calculation, a Fourier-series expression for the output, an IFFT reconstruction over one period, and a Simulink/Simscape comparison against the physical RLC circuit.

\sectionheader{Setup: Circuit and Transfer Function}

\begin{noteBox}
The circuit is a passive third-order RLC lowpass network. The input is the source voltage \(x(t)\). The output \(y(t)\) is measured across the \(100\,\Omega\) load resistor.
\end{noteBox}

\begin{center}

    \node[text=GOLD,font=\large\bfseries] at (6.55,0) {$-$};
\end{circuitikz}

The transfer function is given as
\[
H(s)=\frac{Y(s)}{X(s)}
=\frac{1}{(s/\beta)^3+2(s/\beta)^2+2(s/\beta)+1},
\qquad
\beta=\frac{50}{3}\ \mathrm{rad/s}.
\]
Equivalently,
\[
H(s)=\frac{\beta^3}{s^3+2\beta s^2+2\beta^2s+\beta^3}.
\]
The pulse input has
\[
T_0=1.0\ \mathrm{s},
\qquad
\Omega_0=\frac{2\pi}{T_0}=2\pi\ \mathrm{rad/s},
\qquad
\alpha=0.40.
\]

\sectionheader{Problem (i): Frequency Response with \texttt{polyval}}

\begin{problem}{i}
Using \texttt{polyval}, evaluate the complex frequency response \(H(\jj\Omega)\) for
\[
\Omega=0:(0.1)\Omega_0:(100)\Omega_0.
\]
Plot \(|H(\jj\Omega)|\) in dB and \(\angle H(\jj\Omega)\). Confirm that \(\Omega=\beta=50/3\ \mathrm{rad/s}\) is the \(-3\,\mathrm{dB}\) point.
\end{problem}

\begin{solution}{i}
Substitute \(s=\jj\Omega\):
\[
H(\jj\Omega)
=\frac{\beta^3}{(\jj\Omega)^3+2\beta(\jj\Omega)^2+2\beta^2(\jj\Omega)+\beta^3}.
\]
The magnitude in decibels is
\[
|H(\jj\Omega)|_{\mathrm{dB}}
=20\log_{10}|H(\jj\Omega)|.
\]
At \(\Omega=\beta\), let \(u=\Omega/\beta=1\). Then
\[
H(\jj\beta)
=\frac{1}{(\jj)^3+2(\jj)^2+2\jj+1}.
\]
Since \(\jj^2=-1\) and \(\jj^3=-\jj\),
\[
(\jj)^3+2(\jj)^2+2\jj+1
=-\jj-2+2\jj+1
=-1+\jj.
\]
Therefore
\[
|H(\jj\beta)|
=\frac{1}{|-1+\jj|}
=\frac{1}{\sqrt{2}}.
\]
Thus
\[
20\log_{10}\left(\frac{1}{\sqrt{2}}\right)
=-3.0103\ \mathrm{dB}.
\]
Hence
\[
\boxed{\Omega=\beta=\frac{50}{3}\ \mathrm{rad/s}\text{ is the }-3\ \mathrm{dB}\text{ point}.}
\]
\end{solution}

\sectionheader{Problem (ii): Output Fourier Series}

\begin{problem}{ii}
Assuming unit pulse amplitude and even symmetry about \(t=0\), the input has complex Fourier series
\[
x(t)=\sum_{k=-\infty}^{\infty}X_k e^{\jj k\Omega_0t},
\qquad
X_k=\frac{\sin(k\alpha\pi)}{k\pi}.
\]
Write the output Fourier series and express it as a sum of cosines over nonnegative harmonic indices.
\end{problem}

\begin{solution}{ii}
For \(k=0\), the coefficient is defined by the limiting value
\[
X_0=\lim_{k\to 0}\frac{\sin(k\alpha\pi)}{k\pi}=\alpha.
\]
Because the system is linear time-invariant, each harmonic is multiplied by the frequency response at that harmonic:
\[
Y_k=X_kH(\jj k\Omega_0).
\]
Therefore the output is
\[
\boxed{
y(t)=\sum_{k=-\infty}^{\infty}X_kH(\jj k\Omega_0)e^{\jj k\Omega_0t}.
}
\]

Since the input is real and even, \(X_k\) is real and \(X_{-k}=X_k\). The output can be written using positive harmonics as
\[
\boxed{
y(t)=X_0H(0)+2\sum_{k=1}^{\infty}X_k|H(\jj k\Omega_0)|
\cos\left(k\Omega_0t+\angle H(\jj k\Omega_0)\right).
}
\]
Since \(H(0)=1\) and \(X_0=\alpha=0.40\),
\[
\boxed{
y(t)=0.40+2\sum_{k=1}^{\infty}
\frac{\sin(0.40k\pi)}{k\pi}
|H(\jj 2\pi k)|
\cos\left(2\pi kt+\angle H(\jj 2\pi k)\right).
}
\]

\begin{noteBox}
The coefficient \(X_k\) must keep its sign. Replacing \(X_k\) by \(|X_k|\) without adding a corresponding \(\pi\)-phase correction changes the waveform.
\end{noteBox}
\end{solution}

% ============================================================
\sectionheader{Problem (iii): IFFT Reconstruction over One Period}

\begin{problem}{iii}
Because of the steep roll-off of \(|H(\jj\Omega)|\), truncate the sum at \(|k|=100\). Use \texttt{ifft} to compute \(N=1000\) samples of \(y(t)\) over \([0,T_0)\).
\end{problem}

\begin{solution}{iii}
The truncated coefficient sequence is
\[
Y_k=X_kH(\jj k\Omega_0),
\qquad
-100\le k\le 100.
\]
For an \(N\)-point IFFT, the negative harmonics are placed in the high-index FFT bins by
\[
\text{index}=\operatorname{mod}(k,N)+1.
\]
MATLAB's \texttt{ifft} convention includes a factor \(1/N\), so the output must be multiplied by \(N\):
\[
\boxed{y[n]=N\cdot\operatorname{ifft}\{Y[k]\}.}
\]
The sample times are
\[
t_n=\frac{nT_0}{N},
\qquad
n=0,1,\ldots,N-1.
\]
\end{solution}

\sectionheader{Problem (iv): Simscape Electrical Model}

\begin{problem}{iv}
Generate a Simulink/Simscape Electrical model of the circuit. Use a rectangular pulse train with \(T_0=1.0\ \mathrm{s}\) and duty cycle \(\alpha=40\%\). Simulate for \(10T_0\) using a sampling period of \(20.0\ \mathrm{ms}\). Export the input and output to MATLAB and plot them.
\end{problem}

\begin{solution}{iv}
The physical Simscape model should match the circuit topology:
\[
\text{source}\to 9\,\mathrm{H}\to \text{capacitor node}\to 3\,\mathrm{H}\to 100\,\Omega\to \text{ground}.
\]
The capacitor is connected from the middle node to ground:
\[
C=800\,\mu\mathrm{F}=800\times 10^{-6}\ \mathrm{F}.
\]
The resistor output is measured as the voltage across the \(100\,\Omega\) resistor.

Use the following Simscape/Simulink blocks:

Input source & \texttt{Controlled Voltage Source} or \texttt{Pulse Voltage Source}\
Reactive elements & \texttt{Inductor}, \texttt{Capacitor}\
Load & \texttt{Resistor}\
Measurement & \texttt{Voltage Sensor}\
Conversion & \texttt{PS-Simulink Converter}\
Export & \texttt{To Workspace}\
\bottomrule
\end{tabular}
\end{center}

The helper script in the project folder prints the required block parameters and creates the folders for saving the exported data and plots.
\end{solution}

\sectionheader{Problem (v): Comparison of IFFT and Simulink Outputs}

\begin{problem}{v}
Plot two periods of \(y(t)\) from Part (iii) as a solid graph. On the same figure, plot two periods of the simulated output from Part (iv) with discrete markers.
\end{problem}

\begin{solution}{v}
The theoretical result is periodic with period \(T_0\), so two periods are formed by repeating the one-period IFFT output:
\[
\{y[0],\ldots,y[N-1],y[0],\ldots,y[N-1]\}.
\]
For the Simulink data, use the final two periods of the \(10T_0\) simulation to avoid startup transient effects. If the two curves agree, then the frequency-domain Fourier solution and the time-domain circuit simulation are consistent.
\end{solution}
